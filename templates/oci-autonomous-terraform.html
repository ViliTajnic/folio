What's This About?

Why this matters for enterprise development: Most enterprise applications need rock-solid database infrastructure, but setting it up traditionally takes weeks of planning, configuration, and testing. You'll need an SSH key pair to securely connect to your compute instance. Here's how to create one:

On Windows (PowerShell or WSL):

On macOS/Linux:

# Open Terminal

# Display your public key

Modern Python Connection Patterns

For production enterprise applications, here are the connection patterns used in the scripts:

# Enterprise Connection Pool Pattern
import oracledb
from contextlib import contextmanager

class EnterpriseDBManager:
    """Production-ready database connection manager."""

    def __init__(self, config):
        self.config = config
        self.pool = None

    def initialize_pool(self):
        """Create connection pool for high-performance apps."""
        self.pool = oracledb.create_pool(
            user=self.config['user'],
            password=self.config['password'],
            dsn=self.config['dsn'],
            config_dir=self.config['wallet_location'],
            wallet_location=self.config['wallet_location'],
            wallet_password=self.config['wallet_password'],
            min=2,          # Minimum connections
            max=20,         # Maximum connections
            increment=1,    # Connection increment
            threaded=True   # Thread-safe
        )

    @contextmanager
    def get_connection(self):
        """Context manager for safe connection handling."""
        connection = self.pool.acquire()
        try:
            yield connection
        finally:
            self.pool.release(connection)

Complete Script Code Examples

quick_test.py - The Essential Health Check

#!/usr/bin/env python3
"""
Quick Oracle ADB Connection Test
Simple, fast connectivity verification for automation
"""
import oracledb
import getpass
import sys

def quick_test():
    """Perform a quick connection test with minimal output."""
    try:
        # Get credentials securely
        admin_password = getpass.getpass("Enter ADMIN password: ")
        wallet_password = getpass.getpass("Enter wallet password: ")

        # Quick connection test using high-performance service
        conn = oracledb.connect(
            user="ADMIN",
            password=admin_password,
            dsn="pythonadb_high",
            config_dir="/home/opc/wallet",
            wallet_location="/home/opc/wallet",
            wallet_password=wallet_password
        )

        # Simple test query
        cursor = conn.cursor()
        cursor.execute("SELECT 'Connection successful at ' || TO_CHAR(SYSDATE, 'YYYY-MM-DD HH24:MI:SS') FROM dual")
        result = cursor.fetchone()[0]

        cursor.close()
        conn.close()

        print("‚úÖ", result)
        return True

    except oracledb.DatabaseError as e:
        error_obj = e.args[0] if e.args else None
        error_code = error_obj.code if hasattr(error_obj, 'code') else 'Unknown'
        print(f"‚ùå Connection failed (Error {error_code}): {e}")
        return False

    except Exception as e:
        print(f"‚ùå Unexpected error: {e}")
        return False

if __name__ == "__main__":
    success = quick_test()
    sys.exit(0 if success else 1)  # Return proper exit codes
